class New(webapp2.RequestHandler):		
	def post(self):
		self.response.write('<html><body>')
		user_ID = self.response.get('user_ID', DEFAULT_ID)
		html_content = ""
		css_content = ""
		
		documents = Documents(parent=document_key(user_ID))								
		documents.html_content = html_content
		documents.css_content = css_content
        documents.put()		
		
		query_params = urllib.urlencode({'user_ID': userID, 'html_content': html_content,
										'css_content': css_content})
		self.redirect('/edit?' + urllib.urlencode(query_params))
